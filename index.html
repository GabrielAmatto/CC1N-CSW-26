            const resolverEquacao = (a, b, c) => {
    if (a === 0) {
        return "O coeficiente 'a' não pode ser zero em uma equação de 2º grau.";
    }

    // 1. Calcula o Delta
    const delta = Math.pow(b, 2) - (4 * a * c);
    console.log(`Delta: ${delta}`);

    // 2. Verifica se existem raízes reais
    if (delta < 0) {
        return "A equação não possui raízes reais.";
    }

    // 3. Calcula as raízes usando Math.sqrt
    const x1 = (-b + Math.sqrt(delta)) / (2 * a);
    const x2 = (-b - Math.sqrt(delta)) / (2 * a);

    return delta === 0 ? `Raiz única: ${x1}` : `Raízes: x1 = ${x1}, x2 = ${x2}`;
};

// Exemplo de uso: x² - 5x + 6 = 0
console.log(resolverEquacao(1, -5, 6));
